// --- دالة رصد الأيسبرغ (رصد فقط بدون تغيير الإشارة) --- function analyzeIceberg(bids, asks) { let found = []; const threshold = 0.05; if (lastBids.length > 0) { if (parseFloat(bids[0][0]) === parseFloat(lastBids[0][0])) { let d = parseFloat(bids[0][1]) - parseFloat(lastBids[0][1]); if (d > threshold) found.push({p: bids[0][0], m: d * bids[0][0], t: 'Buyer'}); } if (parseFloat(asks[0][0]) === parseFloat(lastAsks[0][0])) { let d = parseFloat(asks[0][1]) - parseFloat(lastAsks[0][1]); if (d > threshold) found.push({p: asks[0][0], m: d * asks[0][0], t: 'Seller'}); } } if (found.length > 0) { updateIceList(found); document.getElementById('iceberg-card').classList.add('ghost-alert'); document.getElementById('ice-status').innerText = "DETECTED"; document.getElementById('ice-status').style.background = "#450a0a"; } else { document.getElementById('iceberg-card').classList.remove('ghost-alert'); document.getElementById('ice-status').innerText = "SCANNING"; document.getElementById('ice-status').style.background = "#064e3b"; } lastBids = bids; lastAsks = asks; } // --- دالة الجدران (هي المسؤولة الوحيدة عن الـ SIGNAL) --- function renderWalls(bids, asks) { let tB = bids.reduce((s,b)=>s+(parseFloat(b[0])*parseFloat(b[1])),0); let tS = asks.reduce((s,a)=>s+(parseFloat(a[0])*parseFloat(a[1])),0); let buyP = (tB/(tB+tS)*100); // تحديث الإشارة بناءً على سيولة الجدران فقط const sig = document.getElementById('sig-box'); if (buyP > 50) { sig.innerText = "🟢 SIGNAL: BULLISH"; sig.className = "sig-bullish"; } else { sig.innerText = "🔴 SIGNAL: BEARISH"; sig.className = "sig-bearish"; } // تحديث الباور بار والنسب document.getElementById('buy-sell-per').innerText = `BUY: ${buyP.toFixed(1)}% | SELL: ${(100-buyP).toFixed(1)}%`; document.getElementById('power-bar').innerText = "🟢".repeat(Math.round(buyP/6.6)) + "🔴".repeat(15-Math.round(buyP/6.6)); // رسم الجدران document.getElementById('s-walls').innerHTML = `
🔴 SELL WALLS:
` + asks.slice(0,3).map(a => `
Vol: ${(a[0]*a[1]/1000000).toFixed(2)}M${(a[0]*a[1]/tS*100).toFixed(1)}%$${parseFloat(a[0]).toLocaleString()}
`).join(''); document.getElementById('b-walls').innerHTML = `
🟢 BUY WALLS:
` + bids.slice(0,3).map(b => `
Vol: ${(b[0]*b[1]/1000000).toFixed(2)}M${(b[0]*b[1]/tB*100).toFixed(1)}%$${parseFloat(b[0]).toLocaleString()}
`).join(''); }
